"use client";

import { useState, useEffect } from "react";
import { useRouter } from "next/navigation";
import { useReport } from "@/context/ReportContext";
import { createClient } from "@/utils/supabase/client";
import { Activity, BarChart3, Target, CheckCircle2 } from "lucide-react";
import Link from "next/link";

export default function DashboardPage() {
  const [loading, setLoading] = useState(false);
  const [product, setProduct] = useState("");
  const [reports, setReports] = useState<any[]>([]);
  
  const router = useRouter();
  const { setActiveReport } = useReport();
  const supabase = createClient();

  // Unified function for generating reports
const handleGenerate = async () => {
  setLoading(true);
  try {
    const { data: { user } } = await supabase.auth.getUser();

    if (!user) {
      alert("Please log in.");
      return;
    }

    const res = await fetch('/api/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        product, 
        userId: user.id
      }),
    });

    const data = await res.json();
    
    if (res.ok) {
      setActiveReport(data);
      router.push(`/dashboard/report/${data.id}`);
    } else {
      alert("Error: " + (data.error || "Failed to generate"));
    }
  } catch (err) {
    console.error(err);
  } finally {
    setLoading(false);
  }
};



  return (
    <div className="min-h-screen bg-black text-zinc-100">
      <header className="border-b border-zinc-800 bg-black/50 backdrop-blur-md sticky top-0 z-50">
        <div className="max-w-7xl mx-auto px-6 h-16 flex items-center justify-between">
          <div className="flex items-center gap-6">
            <span className="text-xl font-bold tracking-tight">Caviti.io</span>
            <nav className="flex items-center gap-4 text-sm text-zinc-400">
              <span className="text-zinc-600">/</span>
              <span>Workspace</span>
              <span className="text-zinc-600">/</span>
              <span className="text-white">Dashboard</span>
            </nav>
          </div>
          <div className="flex items-center gap-6 text-sm">
            <Link href="/dashboard" className="text-white hover:text-zinc-300">Overview</Link>
            <Link href="/dashboard/billing" className="hover:text-zinc-300">Billing</Link>
            <Link href="/" className="text-zinc-500 hover:text-zinc-300">Sign Out</Link>
          </div>
        </div>
      </header>

      <main className="max-w-7xl mx-auto px-6 py-10">
        <section className="grid grid-cols-1 md:grid-cols-3 gap-6 mb-12">
          {[
            { title: "AI Scans Remaining", value: "8 / 10", icon: Activity },
            { title: "Competitor Flaws Isolated", value: "142", icon: Target },
            { title: "Tracked Niches", value: "3", icon: BarChart3 },
          ].map((stat, i) => (
            <div key={i} className="bg-zinc-950 border border-zinc-800 p-6 rounded-xl">
              <div className="flex items-center gap-3 mb-3 text-zinc-400">
                <stat.icon className="w-5 h-5" />
                <span className="text-sm">{stat.title}</span>
              </div>
              <p className="text-3xl font-semibold">{stat.value}</p>
            </div>
          ))}
        </section>

        <section className="bg-zinc-950 border border-zinc-800 p-8 rounded-xl mb-12">
          <h2 className="text-lg font-medium mb-6">Launch New Competitor Audit</h2>
          <div className="flex gap-4">
            <input 
              type="text" 
              placeholder="Enter product name or competitor URL..." 
              value={product}
              onChange={(e) => setProduct(e.target.value)}
              className="flex-1 bg-black border border-zinc-800 rounded-lg px-4 py-3 text-sm focus:outline-none focus:border-zinc-600"
            />
            <button 
              type="button"
              onClick={handleGenerate}
              disabled={loading}
              className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition"
            >
              {loading ? "Generating..." : "Isolate Gaps"}
            </button>
          </div>
        </section>

        <section className="bg-zinc-950 border border-zinc-800 rounded-xl overflow-hidden">
          <div className="p-6 border-b border-zinc-800">
            <h2 className="text-lg font-medium">Recent Reports</h2>
          </div>
          <table className="w-full text-sm text-left">
            <thead className="text-zinc-500 border-b border-zinc-800">
              <tr>
                <th className="px-6 py-4">Project</th>
                <th className="px-6 py-4">Status</th>
                <th className="px-6 py-4 text-right">Action</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-zinc-800">
              {reports.map((report) => (
                <tr key={report.id}>
                  <td className="px-6 py-4">{report.product_name}</td>
                  <td className="px-6 py-4">
                    <span className="text-emerald-500 inline-flex items-center gap-1.5">
                      <CheckCircle2 className="w-4 h-4" />
                      Completed
                    </span>
                  </td>
                  <td className="px-6 py-4 text-right">
                    <Link href={`/dashboard/report/${report.id}`} className="text-zinc-300 hover:text-white">View Analysis</Link>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </section>
      </main>
    </div>
  );
}
